\resheading{RESEARCH PROJECTS}
  \begin{itemize}[leftmargin=*]
    \item \ressubsingleline{Big Data-based Intent Understanding and Intelligent Input for Natural Interaction}{}{}
    \textit{National Key Research and Development Program}
    {\small
      \begin{itemize}
        \item \justify{Devised a non-invasive body-machine interaction interface (SoftBoMI) utilizing soft strain sensors and an inertial measurement unit. This interface captures shoulder movements to generate control commands for assisting devices.}
      \end{itemize}

      \vspace{-3mm}
      \begin{itemize}
        \item \justify{Designed the online parsing, calibration and data visualization software, and built a virtual simulation environment for wheelchair manipulation based on Unity3D;}
      \end{itemize}    

      \vspace{-3mm}
      \begin{itemize}
        \item \justify{Designed a data decoding framework for user intent inference intervention with shared autonomy, which improves the manipulation accuracy and the dynamic performance of the interface.}
      \end{itemize} 
    }

    \item \ressubsingleline{Stroke Rehabilitation Robot}{}{}     
    \textit{National Key Research and Development Program for Intelligent Robots}
    {\small
    \begin{itemize}
      \item \justify{Built a knee exoskeleton robotic system that maps real-time captured gait motion trajectories from the patient's healthy side to the affected side's actuator, aiming to perform induced gait symmetry restoration;}
    \end{itemize}

    \vspace{-3mm}
    \begin{itemize}
      \item \justify{Designed an encoder-decoder algorithmic framework for online learning and reproduction of gait trajectory time series based on robot imitation learning algorithms.}
    \end{itemize}

    \vspace{-3mm}
    \begin{itemize}
      \item \justify{Constructed a probabilistic form of robot gait skill library by collecting standard gait data, aimed at providing machine strategies.}
    \end{itemize}

    \vspace{-3mm}
    \begin{itemize}
      \item \justify{Designed an online validation, disambiguation, and fine-tuning process for user inputs, realized through a shared autonomy framework, aimed at enhancing the safety of the robotic system.}
    \end{itemize}
    }

    \textit{National Natural Science Foundation of China}

      \item \justify{Proposed a human lower limb motion reconstruction method based on sparse inertial sensor nodes based on LSTM neural network.}
